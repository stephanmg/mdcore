175 __attribute__ ((always_inline)) INLINE void potential_eval_expl ( struct potential *p , FPTYPE r2 , FPTYPE *e , FPTYPE *f ) {
176 
177  const FPTYPE isqrtpi = 0.56418958354775628695;
178  const FPTYPE kappa = 3.0;
179  FPTYPE r = sqrt(r2), ir = 1.0 / r, ir2 = ir * ir, ir4, ir6, ir12, t1, t2;
180  FPTYPE ee = 0.0, eff = 0.0;
181 
182  /* Do we have a Lennard-Jones interaction? */
183  if ( p->flags & potential_flag_LJ126 ) {
184 
185  /* init some variables */
186  ir4 = ir2 * ir2; ir6 = ir4 * ir2; ir12 = ir6 * ir6;
187 
188  /* compute the energy and the force */
189  ee = ( p->alpha[0] * ir12 - p->alpha[1] * ir6 );
190  eff = 6.0 * ir * ( -2.0 * p->alpha[0] * ir12 + p->alpha[1] * ir6 );
191 
192  }
193 
194  /* Do we have an Ewald short-range part? */
195  if ( p->flags & potential_flag_Ewald ) {
196 
197  /* get some values we will re-use */
198  t2 = r * kappa;
199  t1 = erfc( t2 );
200 
201  /* compute the energy and the force */
202  ee += p->alpha[2] * t1 * ir;
203  eff += p->alpha[2] * ( -2.0 * isqrtpi * exp( -t2 * t2 ) * kappa * ir - t1 * ir2 );
204 
205  }
206 
207  /* Do we have a Coulomb interaction? */
208  if ( p->flags & potential_flag_Coulomb ) {
209 
210  /* compute the energy and the force */
211  ee += potential_escale * p->alpha[2] * ir;
212  eff += -potential_escale * p->alpha[2] * ir2;
213 
214  }
215 
216  /* store the potential and force. */
217  *e = ee;
218  *f = eff;
219 
220  }
